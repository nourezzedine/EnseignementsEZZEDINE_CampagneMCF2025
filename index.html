    <title>Tableau Interactif - Ezzedine</title>

    <!-- Inclusion de jQuery -->

    <script src="https://code.jquery.com/jquery-3.6.0.min.js"></script>

    <!-- Inclusion de DataTables -->

    <link rel="stylesheet" href="https://cdn.datatables.net/1.13.6/css/jquery.dataTables.min.css">

    <script src="https://cdn.datatables.net/1.13.6/js/jquery.dataTables.min.js"></script>

    <style>

        table {

            width: 100%;

            margin: 20px auto;

            border-collapse: collapse;

        }

        th, td {

            padding: 10px;

            text-align: left;

            border: 1px solid #ddd;

        }

    </style>

</head>

<body>

    <h2 style="text-align: center;">Enseignements 2019-2024 - Nour Ezzedine</h2>

    <table id="myTable" class="display">

        <thead>

            <!-- Les en-têtes de colonnes seront générés dynamiquement -->

        </thead>

        <tbody>

            <!-- Les lignes du tableau seront générées dynamiquement -->

        </tbody>

    </table>


    <script>

        $(document).ready(function() {

            $.get('EzzedineEnseignements20192024.csv', function(data) {

                const rows = data.split('\n');

                const headers = rows[0].split(',');


                // Génération des en-têtes

                let theadHtml = '<tr>';

                headers.forEach(header => {

                    theadHtml += '<th>' + header + '</th>';

                });

                theadHtml += '</tr>';

                $('#myTable thead').html(theadHtml);


                // Génération des lignes du tableau

                let tbodyHtml = '';

                for (let i = 1; i < rows.length; i++) {

                    const cols = rows[i].split(',');

                    if (cols.length > 1) { // éviter les lignes vides

                        tbodyHtml += '<tr>';

                        cols.forEach(col => {

                            tbodyHtml += '<td>' + col + '</td>';

                        });

                        tbodyHtml += '</tr>';

                    }

                }

                $('#myTable tbody').html(tbodyHtml);


                // Initialisation de DataTables

                $('#myTable').DataTable();

            });

        });

    </script>

</body>
